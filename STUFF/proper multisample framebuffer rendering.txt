
// Create multisampled framebuffer
int fb, color. fbtex;
glGenFramebuffersEXT(1, out fb);
glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, fb);
glGenRenderbuffersEXT(1, out color);
glBindRenderbufferEXT(GL_RENDERBUFFER_EXT, color);
glRenderbufferStorageMultisampleEXT(GL_RENDERBUFFER_EXT, 4, GL_RGBA, 800, 600);
glFramebufferRenderbufferEXT(GL_FRAMEBUFFER_EXT, GL_COLOR_ATTACHEMENT0_EXT, GL_RENDERBUFFER_EXT, color);
// Repeat for depth buffer
// Create non-multisampled framebuffer for reading
glGenFramebuffersEXT(1, out fbtex);
glBindFramebufferEXT(GL_FRAMEBUFFER_EXT);
glBindTexture(GL_TEXTURE_2D, tex);
glFramebufferTexture2DEXT(GL_FRAMEBUFFER_EXT, GL_COLOR_ATTACHMENT, GL_TEXTURE_2D, tex, 0);
glBindTexture(GL_TEXTURE_2D, 0);
